/**
 * This class automates a Flipkart product search using Selenium and TestNG.
 * It searches for "iPhone17" in the Flipkart search bar and submits the query.
 * 
 * @author Mano
 */
public class FlipkartSearchTest {

    private WebDriver driver;
    private WebDriverWait wait;

    /**
     * Sets up ChromeDriver, opens Flipkart, and maximizes the browser window.
     */
    @BeforeMethod
    public void setupBrowser() {
        WebDriverManager.chromedriver().setup();
        driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.get("https://www.flipkart.com/");
        wait = new WebDriverWait(driver, Duration.ofSeconds(10));
    }

    /**
     * Searches for "iPhone17" in Flipkart's search bar and triggers the search.
     */
    @Test
    public void searchForIphone17() {
        WebElement searchBox = wait.until(ExpectedConditions
                .visibilityOfElementLocated(By.xpath("//input[@name='q' and @title='Search for Products, Brands and More']")));
        searchBox.sendKeys("iPhone17");

        WebElement searchButton = wait.until(ExpectedConditions
                .elementToBeClickable(By.xpath("//button[@type='submit' and @title='Search for Products, Brands and More']")));
        searchButton.click();
    }

    /**
     * Closes the browser after the test execution.
     */
    @AfterMethod
    public void tearDown() {
        if (driver != null) {
            driver.quit();
        }
    }
}

package week2.day1;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;

import io.github.bonigarcia.wdm.WebDriverManager;
import java.time.Duration;

/**
 * This class automates navigation to the Playwright website and verifies
 * the Installation heading after clicking the "Get started" link.
 * 
 * Author: Mano
 */
public class PlaywrightGetStartedTest {

    public static void main(String[] args) {

        // Setup ChromeDriver
        WebDriverManager.chromedriver().setup();
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();

        // Create Explicit Wait
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));

        // Navigate to Playwright website
        driver.get("https://playwright.dev/");

        // Click on "Get started" link
        WebElement getStartedLink = wait.until(
                ExpectedConditions.elementToBeClickable(By.xpath("//a[normalize-space()='Get started']")));
        getStartedLink.click();

        // Verify the "Installation" heading is visible
        WebElement installationHeading = wait.until(
                ExpectedConditions.visibilityOfElementLocated(By.xpath("//h1[normalize-space()='Installation']")));

        if (installationHeading.isDisplayed()) {
            System.out.println("Test Passed: 'Installation' heading is visible.");
        } else {
            System.out.println("Test Failed: 'Installation' heading is not visible.");
        }

        // Close browser
        driver.quit();
    }
}
